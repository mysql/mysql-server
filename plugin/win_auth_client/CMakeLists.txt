  SET(HEADERS common.h handshake.h)

      log_client.cc common.cc handshake.cc)

  MYSQL_ADD_PLUGIN(authentication_windows_client ${PLUGIN_SOURCES} ${HEADERS}

    MODULE_ONLY COMPONENT SharedLibraries)
